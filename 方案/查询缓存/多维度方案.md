/**
 * 三种实战方案详细对比
 */

class Solution1_RedisOnly {
  /**
   * 方案1：纯Redis缓存列表
   * 
   * 架构：
   * cache:region:US -> [video_ids...]
   * cache:language:en -> [video_ids...]
   * cache:region:US:language:en -> [video_ids...]  <- 预生成热查询
   */

  constructor() {
    this.redis = null;
  }

  async queryVideos(region, language, style) {
    // 问题：维度爆炸
    // 5个维度 * 10值 = 10万个可能的key组合
    // 实际只预热热查询（几百个）
    
    const key = `cache:${region}:${language}:${style}`;
    const cached = await this.redis.get(key);
    
    if (cached) return JSON.parse(cached);
    
    // 缓存未命中，需要数据库查询或SET交集
    // 使用SET交集: O(N*M) N=最小SET的大小，M=SET个数
    const videoIds = await this.redis.sinter(
      `cache:region:${region}`,
      `cache:language:${language}`,
      `cache:style:${style}`
    );
    
    return videoIds;
  }

  优点: [
    "简单",
    "快速（如果是预热的热查询）",
    "无外部依赖"
  ],

  缺点: [
    "维度爆炸：需要预热所有热查询组合",
    "SET交集操作慢（多维度时可能秒级）",
    "复杂查询困难（OR/范围/聚合）",
    "维护成本高（要手动预热缓存）",
    "难以扩展新维度"
  ]
}

class Solution2_ESOnly {
  /**
   * 方案2：纯Elasticsearch查询
   * 
   * 架构：
   * 一份完整的数据，带多个字段索引
   * {
   *   id: "v1",
   *   title: "Avengers",
   *   region: "US",
   *   language: "en",
   *   year: 2023,
   *   style: "action",
   *   quality: "4K",
   *   hotness: 9.5,  // 热度评分
   *   ...
   * }
   */

  constructor() {
    this.es = null;
  }

  async queryVideos(conditions) {
    // 构建复杂查询（支持任意维度组合）
    const query = this._buildQuery(conditions);
    
    const results = await this.es.search({
      index: 'videos',
      body: query
    });

    return results.hits.hits.map(h => h._source);
  }

  _buildQuery(conditions) {
    const must = [];
    const should = [];

    // 支持灵活的AND/OR逻辑
    if (conditions.region) {
      must.push({ match: { region: conditions.region } });
    }
    if (conditions.year_from && conditions.year_to) {
      must.push({
        range: { year: { gte: conditions.year_from, lte: conditions.year_to } }
      });
    }
    if (conditions.styles && conditions.styles.length > 0) {
      // OR逻辑
      conditions.styles.forEach(style => {
        should.push({ match: { style } });
      });
    }

    return {
      query: {
        bool: {
          must,
          should,
          minimum_should_match: should.length > 0 ? 1 : 0
        }
      },
      sort: conditions.sortBy ? [{ [conditions.sortBy]: 'desc' }] : [],
      from: conditions.offset || 0,
      size: conditions.limit || 20,
      aggs: {
        by_region: { terms: { field: 'region' } },
        by_style: { terms: { field: 'style' } }
      }
    };
  }

  优点: [
    "灵活（支持任意维度任意组合）",
    "强大（全文搜索、范围查询、聚合）",
    "无维度爆炸问题",
    "易于扩展（新增维度只需添加字段）",
    "性能稳定（复杂查询也是毫秒级）"
  ],

  缺点: [
    "内存占用大（倒排索引）",
    "写入慢（需要refresh_interval）",
    "学习曲线（需要理解Lucene/倒排索引）",
    "维护复杂（需要调优参数）",
    "实时性一般（准实时，默认1s）"
  ]
}

class Solution3_Hybrid {
  /**
   * 方案3：混合方案（Redis热缓存 + ES数据库）★★★★★ 推荐
   * 
   * 架构：
   * 1. ES 作为数据存储和查询引擎
   * 2. Redis 缓存热查询结果
   * 3. 热数据自动升级到Redis
   * 
   * 流程：
   * 查询 -> 检查Redis热缓存 -> Miss -> ES查询 -> 缓存热结果到Redis
   */

  constructor() {
    this.redis = null;
    this.es = null;
    this.queryHotness = new Map(); // 追踪查询热度
  }

  async queryVideos(conditions) {
    const cacheKey = this._generateCacheKey(conditions);
    
    // 步骤1：查Redis热缓存
    const cached = await this.redis.get(cacheKey);
    if (cached) {
      this._recordQueryHotness(cacheKey, 'hit');
      return JSON.parse(cached);
    }

    // 步骤2：ES查询
    const results = await this._esQuery(conditions);

    // 步骤3：判断是否为热查询，缓存到Redis
    this._recordQueryHotness(cacheKey, 'miss');
    const hotness = this.queryHotness.get(cacheKey) || 0;
    
    if (hotness > 10) { // 被查询超过10次
      const ttl = 1800; // 30分钟
      await this.redis.setex(cacheKey, ttl, JSON.stringify(results));
    }

    return results;
  }

  _generateCacheKey(conditions) {
    return `query:${JSON.stringify(conditions)}`;
  }

  _recordQueryHotness(cacheKey, type) {
    const current = this.queryHotness.get(cacheKey) || 0;
    this.queryHotness.set(cacheKey, current + 1);
  }

  async _esQuery(conditions) {
    // 使用ES强大的查询能力
    return await this.es.search({
      index: 'videos',
      body: { /* ... */ }
    });
  }

  优点: [
    "兼顾Redis速度和ES灵活性",
    "热数据毫秒级（Redis）",
    "冷查询也可以接受（ES）",
    "自动热数据检测和升级",
    "无维度爆炸问题",
    "易于维护（Redis只缓存热数据）"
  ],

  缺点: [
    "两个系统维护成本",
    "需要缓存失效策略",
    "稍微复杂"
  ]
}
